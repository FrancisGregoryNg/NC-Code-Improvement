18function [] = main(varargin)
19if nargin < 2
20    error('Not enough input arguments!');
21end
22
23inputFile = varargin{1};
24outputFile = varargin{2};
25
26[codeArray] = accessFile(inputFile);
27[codeArray] = deleteComments(codeArray);
28[line] = segregateLines(codeArray);
29[Z,height,neutral,depth] = getZValues(line);
30[plungeRate,feedRate] = getFValues(line);
31[line] = completeLines(line);
32[block] = generateBlocks(line,height,depth);
33[distance,globalConnectionList] = calculateDistances(block);
34[distance] = cleanDistanceMatrix(distance,block);
35[blockConnections] = blockDistance(block,distance);
36[finalSequence] = optimize(blockConnections,block,distance);
37[actualConnections] = rectify(finalSequence,blockConnections,block,distance);
38[finalCode] = translator(finalSequence,actualConnections,block,line,plungeRate,feedRate,height,neutral);
39file = fopen(outputFile, 'w');
40fprintf(file, finalCode);
41fclose('all');
42
43end
