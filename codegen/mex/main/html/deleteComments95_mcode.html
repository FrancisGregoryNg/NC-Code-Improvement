18function [codeArray] = deleteComments(codeArray)
19%{
20Objective: Delete the comments from the code array to make it manageable
21    for subsequent operations on the code. 
22  
23Input: 
24    codeArray = cell array of strings that each contains lines of code
25Output:     
26    codeArray = codeArray without the comments
27%}
28%% Delete Comments
29n=1;
30while n<=length(codeArray)
31    x=1;
32    while x<=length(codeArray{n})
33      if codeArray{n}(x)==';', codeArray{n}(x:length(codeArray{n}))='';end %the semicolon signifies that the rest of the line are just comments
34      x=x+1;
35    end
36    n=n+1;
37end
38end
