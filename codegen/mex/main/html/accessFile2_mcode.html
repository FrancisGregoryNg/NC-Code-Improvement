18function [codeArray] = accessFile(file)
19%{
20Objective: Import the data from the NC file.
21
22Input:      
23    file = filename of the NC file, including the filetype (.nc)
24Output:     
25    codeArray = cell array of strings that each contains lines of code
26%}
27%% Import Data
28%codeArray = importdata(file);
29
30fileID = fopen(file);
31
32rawCode = fread(fileID);
33
34numOfLines = 0;
35for i = 1:length(rawCode)
36    if rawCode(i) == 10
37        numOfLines = numOfLines + 1;
38    end
39end
40
41codeChars = char(numOfLines);
42
43rowNumber = 1;
44colNumber = 1;
45for i = 1:length(rawCode)
46    if rawCode(i) == 13
47        colNumber = 1;
48    elseif rawCode(i) == 10
49        rowNumber = rowNumber + 1;
50    else
51        codeChars(rowNumber, colNumber) = cast(rawCode(i), 'char');
52        colNumber = colNumber + 1;
53    end
54end
55
56codeArray = cell(size(codeChars, 1), 1);
57for i = 1:size(codeChars, 1)
58    codeArray{i} = codeChars(i, :);
59end
60%{
61tline = fgets(fileID);
62codeArray = [];
63while ischar(tline)
64    codeArray = [codeArray; cellstr(tline)];
65    tline = fgets(fileID);
66end
67%}
68end
